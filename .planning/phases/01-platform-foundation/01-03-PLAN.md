  **2. Loop struct:**
  ```
  pub struct Loop {
      config: LoopConfig
      running: bool
      accumulator: u64
      last_time: u64
      fixed_step_ns: u64

      pub func create(config: LoopConfig) Loop {
          const step_ns = cast(u64, 1_000_000_000.0 / config.fixed_hz)
          return Loop {
              config: config,
              running: false,
              accumulator: 0,
              last_time: 0,
              fixed_step_ns: step_ns
          }
      }

      pub func run(self: &Loop) void {
          self.running = true
          self.last_time = getTicksNS()
          self.accumulator = 0

          while(self.running) {
              const now = getTicksNS()
              var frame_time = now - self.last_time
              // Clamp to 250ms to prevent spiral of death (Pitfall 5)
              if(frame_time > 250_000_000) {
                  frame_time = 250_000_000
              }
              self.last_time = now
              self.accumulator = self.accumulator + frame_time

              // Poll events
              while(true) {
                  const ev = pollEvent()
                  if(ev is null) { break }
                  // Check for quit/close — stop the loop
                  if(ev is QuitEvent) { self.running = false }
                  if(ev is WindowCloseEvent) { self.running = false }
                  // Forward to user callback
                  self.config.on_event(ev)
              }

              // Fixed-timestep update
              const dt_seconds = cast(f64, self.fixed_step_ns) / 1_000_000_000.0
              while(self.accumulator >= self.fixed_step_ns) {
                  self.config.on_update(dt_seconds)
                  self.accumulator = self.accumulator - self.fixed_step_ns
              }

              // Variable render with interpolation alpha
              const alpha = cast(f64, self.accumulator) / cast(f64, self.fixed_step_ns)
              self.config.on_render(alpha)
          }
      }

      pub func stop(self: &Loop) void {
          self.running = false
      }

      pub func destroy(self: &Loop) void {
          // Clean state reset (per D-11, LOOP-03)
          self.running = false
          self.accumulator = 0
          self.last_time = 0
      }
  }
  ```

  **Important Orhon syntax notes:**
  - Use `cast(TargetType, value)` for type conversions (confirmed Orhon syntax)
